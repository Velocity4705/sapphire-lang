TECHNICAL DESIGN
================================================================================

1. SPM Command Structure
```bash
spm init [name]           # Initialize new project
spm build                 # Build project
spm run                   # Build and run
spm test                  # Run tests
spm clean                 # Clean build artifacts
spm doc                   # Generate documentation
spm --version             # Show version
spm --help                # Show help
```

2. spm.toml Format
```toml
[package]
name = "my-project"
version = "0.1.0"
authors = ["Your Name <you@example.com>"]
description = "A Sapphire project"

[dependencies]
# Local dependencies for now
# stdlib = { path = "../stdlib" }

[build]
target = "executable"  # or "library"
optimization = "2"

[test]
test_dir = "tests"
```

3. Project Structure
```
my-project/
  spm.toml              # Package manifest
  src/
    main.spp            # Main source file
    lib.spp             # Library code
  tests/
    test_main.spp       # Tests
  docs/
    README.md           # Documentation
  build/                # Build output
    debug/
    release/
```

4. Testing Framework
```sapphire
# Test file: tests/test_math.spp

test "addition works" {
    assert_eq(2 + 2, 4)
}

test "subtraction works" {
    assert_eq(5 - 3, 2)
}

test_group "arithmetic" {
    test "multiplication" {
        assert_eq(3 * 4, 12)
    }
    
    test "division" {
        assert_eq(10 / 2, 5)
    }
}
```

5. Documentation Comments
```sapphire
/// Calculates the factorial of a number
/// 
/// # Arguments
/// * `n` - The number to calculate factorial for
/// 
/// # Returns
/// The factorial of n
/// 
/// # Examples
/// ```
/// let result = factorial(5)
/// assert_eq(result, 120)
/// ```
fn factorial(n: int) -> int {
    if n <= 1 {
        return 1
    }
    return n * factorial(n - 1)
}
```

SPM IMPLEMENTATION

1. Main Entry Point
```cpp
// tools/spm/main.cpp
int main(int argc, char** argv) {
    if (argc < 2) {
        print_help();
        return 1;
    }
    
    std::string command = argv[1];
    
    if (command == "init") {
        return cmd_init(argc - 2, argv + 2);
    } else if (command == "build") {
        return cmd_build(argc - 2, argv + 2);
    } else if (command == "run") {
        return cmd_run(argc - 2, argv + 2);
    } else if (command == "test") {
        return cmd_test(argc - 2, argv + 2);
    } else if (command == "clean") {
        return cmd_clean(argc - 2, argv + 2);
    } else if (command == "doc") {
        return cmd_doc(argc - 2, argv + 2);
    } else if (command == "--version") {
        print_version();
        return 0;
    } else if (command == "--help") {
        print_help();
        return 0;
    } else {
        std::cerr << "Unknown command: " << command << std::endl;
        print_help();
        return 1;
    }
}
```

2. Init Command
```cpp
// tools/spm/commands/init.cpp
int cmd_init(int argc, char** argv) {
    std::string project_name = "my-project";
    if (argc > 0) {
        project_name = argv[0];
    }
    
    // Create project directory
    fs::create_directory(project_name);
    fs::create_directory(project_name + "/src");
    fs::create_directory(project_name + "/tests");
    fs::create_directory(project_name + "/docs");
    
    // Create spm.toml
    create_spm_toml(project_name);
    
    // Create main.spp
    create_main_file(project_name);
    
    // Create README.md
    create_readme(project_name);
    
    std::cout << "Created project: " << project_name << std::endl;
    return 0;
}
```

3. Build Command
```cpp
// tools/spm/commands/build.cpp
int cmd_build(int argc, char** argv) {
    // Load spm.toml
    Project project = load_project("spm.toml");
    
    // Create build directory
    fs::create_directory("build");
    
    // Compile source files
    std::vector<std::string> sources = find_sources("src");
    
    for (const auto& source : sources) {
        std::cout << "Compiling " << source << "..." << std::endl;
        
        std::string cmd = "sapp compile " + source + 
                         " -o build/" + project.name;
        
        int result = system(cmd.c_str());
        if (result != 0) {
            std::cerr << "Compilation failed!" << std::endl;
            return 1;
        }
    }
    
    std::cout << "Build complete: build/" << project.name << std::endl;
    return 0;
}
```

4. Test Command
```cpp
// tools/spm/commands/test.cpp
int cmd_test(int argc, char** argv) {
    // Find test files
    std::vector<std::string> tests = find_tests("tests");
    
    int passed = 0;
    int failed = 0;
    
    for (const auto& test : tests) {
        std::cout << "Running " << test << "..." << std::endl;
        
        // Compile and run test
        std::string cmd = "sapp " + test;
        int result = system(cmd.c_str());
        
        if (result == 0) {
            passed++;
            std::cout << "  ✓ PASSED" << std::endl;
        } else {
            failed++;
            std::cout << "  ✗ FAILED" << std::endl;
        }
    }
    
    std::cout << "\nTests: " << passed << " passed, " 
              << failed << " failed" << std::endl;
    
    return failed == 0 ? 0 : 1;
}
```
