class String {

    size_t capacity_;

    ~String();
    
    size_t length() const;

    String operator+(const String& other) const;
    char operator[](size_t index) const;

    String to_lower() const;

};

2. Vec<T> Dynamic Array
```cpp
template<typename T>
class Vec {
private:
    T* data_;
    size_t size_;
    size_t capacity_;
    
public:
    Vec();
    ~Vec();
    
    void push(const T& value);
    T pop();
    void insert(size_t index, const T& value);
    void remove(size_t index);
    
    T& operator[](size_t index);
    const T& operator[](size_t index) const;
    
    size_t size() const;
    size_t capacity() const;
    bool empty() const;
    
    void clear();
    void reserve(size_t capacity);
    void resize(size_t size);
    
    // Iteration
    T* begin();
    T* end();
};
```

template<typename K, typename V>
class HashMap {
private:
    struct Entry {
        K key;
        V value;
        Entry* next;
    };
    
    Entry** buckets_;
    size_t size_;
    size_t capacity_;
    
    size_t hash(const K& key) const;
    
public:
    HashMap();
    ~HashMap();

    V* get(const K& key);
    bool contains(const K& key) const;

    size_t size() const;
    bool empty() const;
    void clear();

    };
    
    Iterator begin();
    Iterator end();

1. Standard Library Linking
```cpp
// In compiler
void link_stdlib() {
    // Link string functions
    link_function("String_new", stdlib_string_new);
    link_function("String_concat", stdlib_string_concat);
    
    // Link vec functions
    link_function("Vec_new", stdlib_vec_new);
    link_function("Vec_push", stdlib_vec_push);
    
    // Link hashmap functions
    link_function("HashMap_new", stdlib_hashmap_new);
    link_function("HashMap_insert", stdlib_hashmap_insert);
    
    // Link file functions
    link_function("File_open", stdlib_file_open);
    link_function("File_read", stdlib_file_read);
    
    // Link math functions
    link_function("Math_sin", stdlib_math_sin);
    link_function("Math_cos", stdlib_math_cos);
}
```
