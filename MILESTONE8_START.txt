2. Reflection API
```sapphire
# Type reflection
let t = typeof(42)
print(t.name())        # "int"
print(t.size())        # 8
print(t.is_numeric())  # true

# Object reflection
struct Point {
    x: int
    y: int
}

let p = Point { x: 10, y: 20 }
let fields = reflect(p).fields()
for field in fields {
    print(field.name())
    print(field.value())
}
```

3. Compile-Time Constants
```sapphire
# Compile-time constants
const PI = 3.14159
const MAX_SIZE = 1000

# Compile-time evaluation
const RESULT = compute_at_compile_time(10, 20)

# Const functions
const fn factorial(n: int) -> int {
    if n <= 1 { return 1 }
    return n * factorial(n - 1)
}

const FACT_5 = factorial(5)  # Evaluated at compile time
```

REFLECTION SYSTEM

Type Information:
```cpp
class TypeInfo {
private:
    std::string name_;
    size_t size_;
    TypeKind kind_;
    
public:
    std::string name() const;
    size_t size() const;
    TypeKind kind() const;
    bool is_numeric() const;
    bool is_pointer() const;
    bool is_struct() const;
};
```

Reflection API:
```cpp
class Reflection {
public:
    static TypeInfo typeof(const Value& value);
    static std::vector<FieldInfo> fields(const Value& value);
    static Value get_field(const Value& value, const std::string& name);
    static void set_field(Value& value, const std::string& name, const Value& val);
};
```
