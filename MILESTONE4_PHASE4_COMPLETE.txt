Example Usage:
```cpp
// File handle with RC - closes immediately
Rc<File> file = Rc::make<File>("data.txt");
// ... use file ...
// File closes here when RC drops to 0

// Thread-safe Arc
Arc<Connection> conn = Arc::make<Connection>("localhost");
std::thread t1([conn]() { conn->send("hello"); });
std::thread t2([conn]() { conn->send("world"); });
t1.join();
t2.join();
// Connection closes when all threads done

// Weak reference breaks cycle
Rc<Node> a = Rc::make<Node>(1);
Rc<Node> b = Rc::make<Node>(2);
a->next = b;
b->prev = new Weak<Node>(a);  // Weak breaks cycle
// No leak!
```
