1. Allocation Tracking
```cpp
struct AllocationInfo {
    void* ptr;
    size_t size;
    uint64_t type_id;
    uint64_t timestamp;
    bool is_freed;
};

class AllocationTracker {
private:
    std::unordered_map<void*, AllocationInfo> allocations_;
    std::mutex mutex_;
    
public:
    void track_allocation(void* ptr, size_t size, uint64_t type_id);
    void track_free(void* ptr);
    
    size_t total_allocated() const;
    size_t total_freed() const;
    size_t current_usage() const;
    size_t allocation_count() const;
    size_t free_count() const;
    
    std::vector<AllocationInfo> get_live_allocations() const;
    std::vector<AllocationInfo> get_leaks() const;
};
```

2. Memory Statistics
```cpp
struct MemoryStats {
    size_t total_allocated;
    size_t total_freed;
    size_t current_usage;
    size_t peak_usage;
    size_t allocation_count;
    size_t free_count;
    size_t leak_count;
    
    // By size class
    std::map<size_t, size_t> allocations_by_size;
    
    // By type
    std::map<uint64_t, size_t> allocations_by_type;
};

class MemoryProfiler {
public:
    static void enable();
    static void disable();
    static bool is_enabled();
    
    static MemoryStats get_stats();
    static void print_stats();
    static void print_leaks();
    
    static void reset();
};
```

3. Leak Detection

class LeakDetector {

4. Integration with Allocator
```cpp
void* allocate_tracked(size_t size, uint64_t type_id) {
    void* ptr = sapphire_allocate(size, type_id);
    if (MemoryProfiler::is_enabled()) {
        AllocationTracker::track_allocation(ptr, size, type_id);
    }
    return ptr;
}

void free_tracked(void* ptr) {
    if (MemoryProfiler::is_enabled()) {
        AllocationTracker::track_free(ptr);
    }
    sapphire_free(ptr);
}
```

5. Reporting
```cpp
void MemoryProfiler::print_stats() {
    MemoryStats stats = get_stats();
    
    std::cout << "Memory Statistics:\n";
    std::cout << "  Total allocated: " << stats.total_allocated << " bytes\n";
    std::cout << "  Total freed: " << stats.total_freed << " bytes\n";
    std::cout << "  Current usage: " << stats.current_usage << " bytes\n";
    std::cout << "  Peak usage: " << stats.peak_usage << " bytes\n";
    std::cout << "  Allocations: " << stats.allocation_count << "\n";
    std::cout << "  Frees: " << stats.free_count << "\n";
    std::cout << "  Leaks: " << stats.leak_count << "\n";
    
    std::cout << "\nAllocations by size:\n";
    for (const auto& [size, count] : stats.allocations_by_size) {
        std::cout << "  " << size << " bytes: " << count << " allocations\n";
    }
    
    std::cout << "\nAllocations by type:\n";
    for (const auto& [type_id, count] : stats.allocations_by_type) {
        std::cout << "  Type " << type_id << ": " << count << " allocations\n";
    }
}
```

================================================================================
LEAK DETECTION STRATEGY
================================================================================

Simple Leak Detection:
1. Track all allocations
2. Track all frees
3. At end of program, check for unfreed allocations
4. Report leaks with size and type

Example:
```cpp
void check_leaks() {
    auto leaks = LeakDetector::find_leaks();
    
    if (leaks.empty()) {
        std::cout << "No memory leaks detected! ✅\n";
    } else {
        std::cout << "Memory leaks detected: " << leaks.size() << " ❌\n";
        for (void* ptr : leaks) {
            AllocationInfo info = LeakDetector::get_leak_info(ptr);
            std::cout << "  Leak at " << ptr << ": " 
                     << info.size << " bytes, type " << info.type_id << "\n";
        }
    }
}
```

PERFORMANCE CONSIDERATIONS
================================================================================

Profiling Overhead:
- Tracking allocation: ~50-100 cycles
- Tracking free: ~50-100 cycles
- Hash map lookup: ~20-50 cycles
- Mutex lock: ~20-50 cycles

Optimization Strategies:
1. Only enable when needed
2. Use thread-local tracking
3. Batch updates
4. Minimize locking

Expected Overhead:
- Disabled: 0% (no overhead)
- Enabled: ~10-20% (acceptable for debugging)
- Target: < 20% when enabled ✅
