================================================================================
EXAMPLE TRANSFORMATION
================================================================================

Sapphire Code:
--------------
let x = 10
let y = 20
let sum = x + y
print(sum)

LLVM IR (to be generated):
---------------------------
define i32 @main() {
entry:
  %x = alloca i64
  store i64 10, i64* %x
  %y = alloca i64
  store i64 20, i64* %y
  %0 = load i64, i64* %x
  %1 = load i64, i64* %y
  %sum = add i64 %0, %1
  %2 = call i32 @print_i64(i64 %sum)
  ret i32 0
}

Native Assembly (x86_64):
-------------------------
main:
  mov rax, 10
  mov rbx, 20
  add rax, rbx
  mov rdi, rax
  call print_i64
  xor eax, eax
  ret
