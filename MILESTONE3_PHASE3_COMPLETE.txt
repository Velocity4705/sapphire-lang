================================================================================
EXAMPLE: GENERATED IR
================================================================================

Input (examples/expr_test.spp):
```sapphire
let a = 10
let b = 20
let sum = a + b
let greater = a > b
```

Output (LLVM IR):
```llvm
define i32 @main() {
entry:
  %a = alloca i64, align 8
  store i64 10, ptr %a, align 4
  %b = alloca i64, align 8
  store i64 20, ptr %b, align 4
  %a1 = load i64, ptr %a, align 4
  %b2 = load i64, ptr %b, align 4
  %addtmp = add i64 %a1, %b2
  %sum = alloca i64, align 8
  store i64 %addtmp, ptr %sum, align 4
  %a3 = load i64, ptr %a, align 4
  %b4 = load i64, ptr %b, align 4
  %cmptmp = icmp sgt i64 %a3, %b4
  %greater = alloca i1, align 1
  store i1 %cmptmp, ptr %greater, align 1
  ret i32 0
}
```

✅ Valid LLVM IR
✅ Proper types (i64, i1, double)
✅ SSA form
✅ Ready for optimization
