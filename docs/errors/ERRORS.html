E1xx

Syntax

E1A0unexpected token
note: check for missing colons, parentheses, or mismatched brackets
# bad
if x > 0
    print("yes")

# fix
if x > 0:
    print("yes")
E1A1expected expression
note: an expression (value, variable, or function call) was expected here
# bad
x = +

# fix
x = 1 + 2
E1A2invalid indentation
note: Sapphire uses 4-space indentation — check for mixed tabs/spaces
# bad — tab used
fn greet():
	print("hi")

# fix — 4 spaces
fn greet():
    print("hi")
E1A3unterminated string
note: add a closing quote to end the string literal
name = "Alice   # missing closing "
name = "Alice"
E1A4invalid decorator placement
note: decorators can only be applied to functions and classes
@cached
x = 10

# fix
@cached
fn compute():
    ...
E1A5missing colon
note: fn, class, if, for, and while bodies must end with ':'
fn add(a, b)
    return a + b

fn add(a, b):
    return a + b
E1A6invalid assignment target
note: only variables and subscript expressions can be assigned to
1 = x
x = 1
E1A7body must be indented on the next line
note: Sapphire does not allow single-line bodies — put the body on the next line, indented. Applies to fn, class, if, while, and for.
if x > 0: print("yes")

if x > 0:
    print("yes")
E2xx

Name

E2A0undefined variable
note: check the spelling, or make sure the variable is defined before use
print(count)   # count not defined

count = 0
print(count)
E2A1undefined function
note: check the function name, or make sure the module is imported
result = sqrt(16)

import math
result = math.sqrt(16)
E2A2undefined module
note: check the module name — use 'import <module>' at the top of the file
import mathz   # typo
import math
E2A3attribute not found
note: check the attribute name, or verify the object type
x = 42
x.append(1)   # int has no append

items = []
items.append(1)
E3xx

Type

E3A0type mismatch
note: the value type does not match what was expected
x = "hello" + 5
x = "hello" + str(5)
E3A1unsupported operand types
note: this operator cannot be used with these types
result = [1, 2] * [3, 4]
result = [1, 2] + [3, 4]   # concatenation
E3A2not callable
note: only functions and classes can be called with ()
x = 42
x()

fn greet():
    print("hi")
greet()
E3A3not iterable
note: for loops require a list, string, range, or other iterable
for i in 42:
    print(i)

for i in range(42):
    print(i)
E3A4not subscriptable
note: only lists, strings, and dicts support [] indexing
x = 42
print(x[0])

items = [42]
print(items[0])
E4xx

Runtime

E4A0index out of bounds
note: the index is outside the valid range — check the list length first
items = [1, 2, 3]
print(items[5])

if 5 < len(items):
    print(items[5])
E4A1division by zero
note: the divisor evaluated to zero — add a check before dividing
result = 10 / 0

divisor = 4
if divisor != 0:
    result = 10 / divisor
E4A2assertion failed
note: an assert() condition was false — check the logic being asserted
assert(1 == 2)
assert(len(items) > 0)
E4A3null dereference
note: the value is 'none' — check for none before accessing it
x = none
print(x.name)

if x != none:
    print(x.name)
E4A4stack overflow
note: infinite or deeply recursive function — check your base case
fn count(n):
    return count(n - 1)   # no base case

fn count(n):
    if n <= 0:
        return 0
    return count(n - 1)
E5xx

Import

E5A0module not found
note: check the module name spelling and that it is installed
import nonexistent_module

import math
import json
import http
E5A1circular import
note: two modules are importing each other — restructure to break the cycle
# a.spp imports b.spp
# b.spp imports a.spp  ← cycle

# fix: extract shared code into c.spp
# both a.spp and b.spp import c.spp
E6xx

IO

E6A0file not found
note: check the file path and that the file exists
content = fs.read("data.txt")   # file missing

import fs
if fs.exists("data.txt"):
    content = fs.read("data.txt")
E6A1permission denied
note: the process does not have permission to access this file
# Linux/macOS — fix file permissions:
# chmod 644 file.txt
E7xx

Value

E7A0invalid argument
note: a function received an argument with an unexpected value
import math
math.sqrt(-1)

import math
math.sqrt(abs(-1))
E7A1argument count mismatch
note: check the number of arguments the function expects
fn add(a, b):
    return a + b

add(1)        # too few
add(1, 2, 3)  # too many
add(1, 2)     # correct
E7A2value out of range
note: the value exceeds the allowed range for this operation
import time
time.sleep(-5)
time.sleep(1)
